${i+1}. ${item.q}
`; const opts = document.createElement('div'); opts.className = 'options'; item.opts.forEach((o, idx)=>{ const b = document.createElement('div'); b.className = 'opt'; b.textContent = o; b.addEventListener('click', ()=>{ // mark selection const siblings = opts.querySelectorAll('.opt'); siblings.forEach(s=>s.classList.remove('selected')); b.classList.add('selected'); answers[i] = idx; // store index }); // preselect if already answered if(typeof answers[i] !== 'undefined' && answers[i] === idx) b.classList.add('selected'); opts.appendChild(b); }); qWrap.appendChild(opts); quizArea.appendChild(qWrap); // show/hide prev prevBtn.style.display = i===0? 'none' : 'inline-block'; nextBtn.textContent = i===questions.length-1? 'Finish' : 'Next'; } function openQuiz(){ quizSection.style.display = 'block'; resultsSection.style.display = 'none'; current = 0; answers = []; renderQuestion(current); window.scrollTo({top:0,behavior:'smooth'}); } startBtn.addEventListener('click', openQuiz); takeQuizBtn.addEventListener('click', openQuiz); closeQuiz.addEventListener('click', ()=>{quizSection.style.display='none'}); nextBtn.addEventListener('click', ()=>{ // require selection if(typeof answers[current] === 'undefined'){ alert('Please select an option to continue.'); return; } if(current < questions.length-1){ current++; renderQuestion(current); window.scrollTo({top:0,behavior:'smooth'}); } else { // finish quizSection.style.display = 'none'; showResults(); } }); prevBtn.addEventListener('click', ()=>{ if(current>0){current--;renderQuestion(current)} }); function showResults(){ // naive scoring: map answers to clusters const tally = {dev:0,design:0,people:0,data:0}; answers.forEach((ans, i)=>{ // map question choices to clusters const mapping = [ [ 'data','design','people','dev' ], [ 'data','design','people','dev' ], [ 'dev','design','people','dev' ], [ 'data','design','people','dev' ], [ 'data','design','people','dev' ], [ 'dev','data','design','people' ] ]; const key = mapping[i][ans]; tally[key]++; }); // pick top 2 const sorted = Object.entries(tally).sort((a,b)=>b[1]-a[1]); const primary = sorted[0][0]; const secondary = sorted[1][0]; const suggestions = { dev: {title:'Software Developer', steps:['Learn JS fundamentals','Build 3 small projects','Learn Git & deploy'], skills:['JavaScript','HTML','CSS']}, data: {title:'Data Analyst', steps:['Learn SQL basics','Python for data','Practice on datasets'], skills:['SQL','Python','Excel']}, design: {title:'UX / UI Designer', steps:['Learn design basics','Practice Figma','Build a portfolio'], skills:['Figma','Design thinking','Prototyping']}, people: {title:'Product / Marketing', steps:['Learn communication','Study case studies','Run small campaigns'], skills:['Communication','Analytics','Strategy']} }; resultsGrid.innerHTML = ''; const primaryCard = document.createElement('div'); primaryCard.className='result-item'; primaryCard.innerHTML = `
Best match: ${suggestions[primary].title}
Top skills: ${suggestions[primary].skills.join(', ')}
Starter steps
    ${suggestions[primary].steps.map(s=>`
  1. ${s}
    2. `).join('')}
`; const secondaryCard = document.createElement('div'); secondaryCard.className='result-item'; secondaryCard.innerHTML = `
Runner-up: ${suggestions[secondary].title}
Also consider: ${suggestions[secondary].skills.join(', ')}
Quick steps
    ${suggestions[secondary].steps.map(s=>`
  1. ${s}
    2. `).join('')}
`; resultsGrid.appendChild(primaryCard); resultsGrid.appendChild(secondaryCard); // action cards const learnCard = document.createElement('div'); learnCard.className='result-item'; learnCard.innerHTML = `
Learning Path
Curated micro-steps & free resources to get started.
`; resultsGrid.appendChild(learnCard); resultsSection.style.display = 'block'; window.scrollTo({top:0,behavior:'smooth'}); } function downloadPlan(){ // prepare a simple text plan and download as .txt const text = 'Your 30-day starter plan from Futurize 2.0\n\n' + '1) Week 1: Basics...\n2) Week 2: Projects...\n(Use this as a template)'; const blob = new Blob([text], {type:'text/plain'}); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'futurize-plan.txt'; a.click(); URL.revokeObjectURL(url); } // quick onboarding: save profile (localStorage) document.getElementById('save-profile').addEventListener('click', ()=>{ const profile = {answers, date: new Date().toISOString()}; const list = JSON.parse(localStorage.getItem('futurize_profiles')||'[]'); list.unshift(profile); localStorage.setItem('futurize_profiles', JSON.stringify(list)); alert('Profile saved locally.
